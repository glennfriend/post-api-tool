    function createEmail()
    {
        var currentdate = new Date();
        var datetime =
                currentdate.getFullYear() + "_"
                + (currentdate.getMonth()+1)  + "_"
                + currentdate.getDate() + "."
                + currentdate.getHours() + ""
                + currentdate.getMinutes() + ""
                + currentdate.getSeconds();
        return 'just.test.' + datetime + '@local.com';
    }
